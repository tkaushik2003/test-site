pipeline {
    agent none
    stages {
	
	stage('Non-Parallel Stage') {
	    agent {
                        label "main"
                }
        steps {
                echo 'This stage will be executed first'
                }
        }

	
        stage('Run Tests') {
            parallel {
                stage('Test On Windows') {
                    agent {
                        label "Windows_Node"
                    }
                    steps {
                        echo "build successful"
                    }
                    
                }
                stage('Test On Main') {
                    agent {
                        label "main"
                    }
                    steps {
						echo "passed all test cases"
					}
                }
            }
        }
    }
}
